#include<iostream>
using namespace std;
class complex
{
    int real;
    int imag;
    public:
    complex()
    {
    }
    complex(int a)
    {
       real=a;
       imag=a;
    }
    complex(int a,int b)
    {
        real=a;
        imag=b;
    }
    void sum (complex c1,complex c2)
    {
        
        real=c1.real+c2.real;
        imag=c1.imag+c2.imag;
        
    }
    void display()
    {
        if (imag<0)
         cout<<real<<imag<<"i"<<endl;
        else{
        cout<<real<<"+"<<imag<<"i"<<endl;
             }
        
    }
    
};
int main(){
    int r1,i1,r2,i2;
   
cout<<"enter each values for the real and imaginary part of the first complex : ";
cin>>r1>>i1;
cout<<"enter single value for the real and imaginary part of the second complex : ";
cin>>r2;
complex c1(r1,i1);
complex c2(r2);
complex c3;

c3.sum(c1,c2);
cout<<"first complex :";
c1.display();
cout<<"second complex :";
c2.display();
cout<<"resultant complex :";
c3.display();


return 0;
}
