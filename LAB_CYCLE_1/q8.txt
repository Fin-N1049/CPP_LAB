class complex
{
    int real;
    int imag;
    public:
    complex()
    {
    }
    complex(int a)
    {
       real=a;
       imag=a;
    }
    complex(int a,int b)
    {
        real=a;
        imag=b;
    }
    void sum (complex c1,complex c2)
    {

        real=c1.real+c2.real;
        imag=c1.imag+c2.imag;

    }
     void display()
    {
        if (imag<0)
         cout<<real<<imag<<"i"<<endl;

             }
